CREATE OR REPLACE TRIGGER KATEGORIA_INITCAP
BEFORE INSERT ON KATEGORIE
FOR EACH ROW 
BEGIN
    :NEW.NAZWA:=INITCAP(:NEW.NAZWA);
END;
/
ALTER TRIGGER KATEGORIA_INITCAP ENABLE;

create or replace trigger trig_naleznosc
before update on kary_wypozyczenia
for each row
begin
 if :new.naleznosc < 5 then
 :new.naleznosc:= 5;
 end if;
end;
/
ALTER TRIGGER trig_naleznosc ENABLE;
